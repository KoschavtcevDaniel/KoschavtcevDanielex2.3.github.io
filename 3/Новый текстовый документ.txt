CREATE TABLE ability (
 id int(10) unsigned NOT NULL AUTO_INCREMENT,
 title int(4) NOT NULL DEFAULT '',
 PRIMARY KEY (id)
);

CREATE TABLE application_ability (
 id int(10) unsigned NOT NULL AUTO_INCREMENT,
 aplication_id int(10) NOT NULL DEFAULT 0,
 ability_id int(10) NOT NULL DEFAULT 0,
 PRIMARY KEY (id)
);
